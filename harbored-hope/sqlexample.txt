CREATE TABLE dbo.app_users
(
    user_id            INT IDENTITY(1, 1) NOT NULL,
    first_name         NVARCHAR(100) NOT NULL,
    last_name          NVARCHAR(100) NOT NULL,
    email              NVARCHAR(255) NOT NULL,
    password_hash      NVARCHAR(450) NOT NULL,
    role               VARCHAR(20) NOT NULL,
    supporter_id       INT NULL,
    email_confirmed    BIT NOT NULL
        CONSTRAINT DF_app_users_email_confirmed DEFAULT (0),
    created_at         DATETIME2(7) NOT NULL
        CONSTRAINT DF_app_users_created_at DEFAULT (SYSUTCDATETIME()),
    updated_at         DATETIME2(7) NOT NULL
        CONSTRAINT DF_app_users_updated_at DEFAULT (SYSUTCDATETIME()),
    last_login_at      DATETIME2(7) NULL,
    is_active          BIT NOT NULL
        CONSTRAINT DF_app_users_is_active DEFAULT (1),
    row_version        ROWVERSION NOT NULL,
    CONSTRAINT PK_app_users PRIMARY KEY CLUSTERED (user_id),
    CONSTRAINT UQ_app_users_email UNIQUE (email),
    CONSTRAINT CK_app_users_role CHECK (role IN (N'donor', N'admin'))
);

CREATE NONCLUSTERED INDEX IX_app_users_supporter_id
    ON dbo.app_users (supporter_id)
    WHERE supporter_id IS NOT NULL;

CREATE NONCLUSTERED INDEX IX_app_users_role
    ON dbo.app_users (role)
    INCLUDE (email, is_active);
